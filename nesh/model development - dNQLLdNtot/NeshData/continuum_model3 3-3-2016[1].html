# This is the run

# Timing
sec1 = time.time()

# Loop over the time steps
print 'ntimesteps', ntimesteps
ifirst = 0
equilibration_ticker = 0
Ntot = Fliq + Nice
for i in range(ntimesteps):

    # Figure out how much new water to deposit
    delta = (Fliq - (Nbar - Nstar))/(2*Nstar)
    sigD = (sigmastep - delta * sigma0)/(1+delta*sigma0)
    depsurf = deprate_times_deltaT * sigD
    
    # Figure out the change due to diffusion
    delta_diffusion = ds.diffuseconstantDP(Fliq, Dprimesurface)
    
    # Combine deposition and diffusion
    deltaFliq = depsurf + delta_diffusion
    
    # Increment QLL and total accordingly
    Fliq += deltaFliq
    Ntot += deltaFliq

    # Apply the algebraic constraints regarding ice/QLL equilibration
    if (equilibration_ticker < equilibration_ticker_max):
        equilibration_ticker += deltaT
    else:
        Fliq = ds.getNliq(Fliq + Nice,Nstar,Nbar,niter)  
        Nice = Ntot - Fliq      
        equilibration_ticker = 0
    
    # Diagnostics
    sigavgstart.append(sigD[0])
    sigavgend.append(sigD[nmid])
        
# Diagnostics
minpoint = min(Nice)
