# This is the run

# Timing
sec1 = time.time()

# Loop over the time steps
print 'ntimesteps', ntimesteps
ifirst = 0
equilibration_ticker = 0
Ntot = Fliq + Nice
for i in range(ntimesteps):

    # Figure out how much new water to deposit
    delta = (Fliq - (Nbar - Nstar))/(2*Nstar)
    sigD = (sigmastep - delta * sigma0)/(1+delta*sigma0)
    depsurf = deprate_times_deltaT * sigD
    
    # Figure out the change due to diffusion
    delta_diffusion = ds.diffuseconstantDP(Fliq, Dprimesurface)
    
    # Combine deposition and diffusion
    deltaFliq = depsurf + delta_diffusion
    
    # Increment QLL and total accordingly
    Fliq += deltaFliq
    Ntot += deltaFliq

    # Apply the algebraic constraints regarding ice/QLL equilibration
    if (equilibration_ticker < equilibration_ticker_max):
        equilibration_ticker += deltaT
    else:
        Fliq = ds.getNliq(Fliq + Nice,Nstar,Nbar,niter)  
        Nice = Ntot - Fliq      
        equilibration_ticker = 0
    
    # Diagnostics
    sigavgstart.append(sigD[0])
    sigavgend.append(sigD[nmid])

minpoint = min(Nice)

# Hertz-Knudsen stuff

# This formula is from B. J. Mason, The Physics of Clouds (Clarendon Press, 1971)
def getPice(TC):
    TK = TC + 273.16
    return (3.6646e10 - 1.3086e6*TC - 33793*TC**2)*np.exp(-6150/TK)*100

# This is from http://www.conservationphysics.org/atmcalc/atmoclc2.pdf
def getPice2(TC):
    return np.exp( -6140.4 / ( 273 + TC ) + 28.916 ) 

TC = -30
print getPice(TC), 'Pa'
print getPice2(TC), 'Pa'

# A test of Libbrecht's formulas

# Example conditions
TK = TC+273.16
R = 8.314 # J/mol-K
Mice = .018015 # kg/mol

# Mole density of water vapor
csat = getPice(TC)/(R*TK) # P/RT = n/V
print csat, 'mol/m^3'

# Mole density of solid ice
rhoice = 920 # kg/m^3; this is at -30C
#see http://www.engineeringtoolbox.com/ice-thermal-properties-d_576.html
csolid = rhoice/Mice
print csolid, 'mol/m^3'

# Ratio of densities
densityratio = csat/csolid
print densityratio, 'csat/csolid'

# Kinetic velocity according to Libbrecht's Eq. A6
nu_kin_A6 = densityratio*np.sqrt(R*TK/(2*np.pi*Mice))*1e6
print nu_kin_A6, 'um/s'
