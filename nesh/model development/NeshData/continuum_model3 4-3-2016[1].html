count = 0
equilibration_ticker = 0

while Nice0 < nlayers:
    
    # Figure out how much new water to deposit
    delta = (Fliq0 - (Nbar - Nstar))/(2*Nstar)
    sigD = (sigmastepmax - delta * sigma0)/(1+delta*sigma0)
    depsurf = deprate_times_deltaT * sigD

    # Increment QLL and total accordingly
    Fliq0 += depsurf
    Ntot0 += depsurf

    # Apply the algebraic constraints regarding ice/QLL equilibration
    if count%equilibrationfreq == 0:
        Fliq0 = ds.getNliq(Ntot0,Nstar,Nbar,niter)  
        Nice0 = Ntot0 - Fliq0      

    # Increment the time

    count += 1
